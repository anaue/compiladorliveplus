program
    fatorial_num int,
    fatorial_res int
    //fatorial_res int[3]
begin
    fatorial_num = 3;
    fatorial_res = fatorial_num;
    while fatorial_num > 1 loop
        fatorial_num = fatorial_num - 1 ;
        fatorial_res = fatorial_num * fatorial_res ;
    endloop;
    //output fatorial_res ;
end ;
